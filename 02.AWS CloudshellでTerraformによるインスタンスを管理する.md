# `01.AWS CloudshellでTerraformを始める`に引き続きEC2のデプロイまでを行う

```bash
# ==========================================
# 1. 宣言部：Terraform本体とAWSプラグインのバージョン指定
# ========================================== 
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 5.0"
    }
  }
}

# ==========================================
# 2. 接続先（プロバイダー）の指定：東京リージョンへ繋ぐ
# ==========================================
provider "aws" {
  region = "ap-northeast-1"
}

# ==========================================
# 3. 実体（リソース）の作成：ハローワールド用VPC
# ==========================================
resource "aws_vpc" "hello_vpc" {
  cidr_block = "10.0.0.0/24" # 最小限に絞ったCIDRブロック

  tags = {
    Name = "HelloWorld-VPC"
  }
}

# ==========================================
# 4. Subnet (サブネットの作成)
# ==========================================
resource "aws_subnet" "my_subnet" {
  vpc_id                  = aws_vpc.my_vpc.id
  cidr_block              = "10.0.1.0/24"
  availability_zone       = "ap-northeast-1a" # 東京リージョンのAゾーン
  map_public_ip_on_launch = true              # パブリックIPを自動付与（重要）

  tags = {
    Name = "t-consider-subnet"
  }
}

# ==========================================
# 5. Security Group (ファイアウォールの作成)
# ==========================================
resource "aws_security_group" "my_sg" {
  name        = "t-consider-sg"
  description = "Allow SSH inbound traffic"
  vpc_id      = aws_vpc.my_vpc.id

  # インバウンドルール (外からの通信：SSHの22番ポートを許可)
  ingress {
    description = "SSH from anywhere"
    from_port   = 22
    to_port     = 22
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"] # ※学習用なので全開放していますが、本来は自IP等絞る
  }

  # アウトバウンドルール (中からの通信：全て許可)
  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }

  tags = {
    Name = "t-consider-sg"
  }
}

# ==========================================
# 6. Amazon Machine Image (最新のAmazon Linux 2023を自動取得)
# ==========================================
data "aws_ami" "amazon_linux" {
  most_recent = true
  owners      = ["amazon"]

  filter {
    name   = "name"
    values = ["al2023-ami-2023.*-x86_64"]
  }
}

# ==========================================
# 7. EC2 Instance (サーバー本体の作成)
# ==========================================
resource "aws_instance" "my_ec2" {
  ami           = data.aws_ami.amazon_linux.id
  instance_type = "t2.micro" # 無料枠のサイズ
  subnet_id     = aws_subnet.my_subnet.id
  
  # 作成したセキュリティグループを紐付け
  vpc_security_group_ids = [aws_security_group.my_sg.id]

  tags = {
    Name = "t-consider-ec2"
  }
}

```
